# -----------------------------------------------------------------------------
# Enhanced ForestFire : Sparse forest with barriers and varied fire spread
#
# Description :
# - Sparse tree coverage with void areas and natural barriers (rocks).
# - Increased difficulty for fire propagation due to lower density.
# - Fire starts in multiple locations with varied distances.
# -----------------------------------------------------------------------------

world 64 64 FFF        # larger 64x64 grid with white void cells

mineral tree 0F0       # green for 'tree' mineral agents
var hot                # shortcut for "var hot 0 0"
sensor hot flame 1     # 'flame' field detected by 'tree'
status hot < 2 tree    # 'tree' -> 'tree' when 'hot' < 2
status hot > 1 fire    # 'tree' -> 'fire' when 'hot' > 1

mineral fire F00       # red for 'fire' mineral agents
var flame 2            # fire creates 'flame' field
status ash             # 'fire' -> 'ash' unconditionally
field flame -1         # flame reduces by 1 per distance step

mineral ash 777        # gray for 'ash' mineral agents
                       # no transition rules, remains 'ash'

mineral rock 555       # dark gray for 'rock' mineral agents
block flame            # blocks flame propagation

# -----------------------------------------------------------------------------
# initial configuration : sparse forest with barriers
# -----------------------------------------------------------------------------

agent (void,tree,void,tree) (0:64,0:64) # sparse forest with empty patches
agent rock (20:25,20:25) (40:45,40:45) # two barrier zones
agent fire (5,5) (60,60) (30,15)       # multiple initial fire sources
